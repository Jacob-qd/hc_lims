const getStatusColor = (status: string) => {
  const map: Record<string, string> = {
    active: 'success',
    revoked: 'error',
    expired: 'warning',
  };
  return map[status] || 'default';
};
 
const getStatusLabel = (status: string) => {
  const map: Record<string, string> = {
    active: '有效',
    revoked: '已吊销',
    expired: '已过期',
  };
  return map[status] || status;
};
